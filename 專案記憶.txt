### ▶️ `scraper.py` - 低風險单次抓取
```python
import requests, time, random
from bs4 import BeautifulSoup

def fetch_venue_slots(venue_id="352", target_dates=None):
    """
    单次抓取場地時段，加入隨機延遲模擬真人
    返回：list of {date, time, status, booker}
    """
    url = f"https://vbs.sports.taipei/venues/?K={venue_id}"
    headers = {
        "User-Agent": "Mozilla/5.0 (Windows NT 10.0; Win64; x64) AppleWebKit/537.36",
        "Accept-Language": "zh-TW,zh;q=0.9"
    }
    
    # 🛡️ 低風險：隨機延遲 2-5 秒
    time.sleep(random.uniform(2, 5))
    
    try:
        resp = requests.get(url, headers=headers, timeout=15)
        resp.raise_for_status()
        soup = BeautifulSoup(resp.text, 'html.parser')
        
        slots = []
        # ⚠️ 實際選擇器需依網頁結構調整，此為範例
        for item in soup.select('.slot-item'):
            slot = {
                'date': item.get('data-date'),
                'time': item.get('data-time'),
                'status': 'booked' if item.get('data-booked') == '1' else 'available',
                'booker': item.get('data-booker-name', '').strip()
            }
            # 過濾：只保留目標日期 + 台灣網球協會相關
            if target_dates and slot['date'] not in target_dates:
                continue
            if slot['booker'] and '台灣網球協會' in slot['booker']:
                slots.append(slot)
        return slots
    except Exception as e:
        print(f"[抓取錯誤] {e}")
        return []
```

### ▶️ `ai_helper.py` - Gemini 整合
```python
import os
from google import genai
from google.genai import types

# 初始化 Gemini Client (會自動讀取環境變數 GEMINI_API_KEY)
client = genai.Client()

def generate_telegram_reply(slots, user_query):
    """產生適合 Telegram 推送的簡潔回覆"""
    prompt = f"""
    你是網球場租借小助手，用繁體中文、親切簡潔風格回覆。
    
    使用者查詢：{user_query}
    找到 {len(slots)} 筆「台灣網球協會」相關預訂：
    {slots}
    
    請輸出：
    1. 🎾 標題 + 日期範圍
    2. 📋 時段清單（用 emoji + 短條列）
    3. ⚠️ 重要提醒（如：需提前10天申請、僅第6面可線上租）
    4. 💡 下一步建議（如：「回覆 /book 5/15 16:00 可幫你整理申請須知」）
    
    限制：總長度 ≤ 4096 字元（Telegram 限制）
    """
    
    try:
        response = client.models.generate_content(
            model='gemini-2.5-flash',
            contents=prompt,
            config=types.GenerateContentConfig(
                temperature=0.3,
            )
        )
        return response.text
    except Exception as e:
        print(f"[Gemini API 錯誤] {e}")
        return "抱歉，目前無法產生彙整回覆，請稍後再試。🎾"
```

### ▶️ `bot.py` - Telegram Bot 主入口
```python
from telegram import Update
from telegram.ext import Application, CommandHandler, ContextTypes
import scraper, ai_helper
import os

BOT_TOKEN = os.environ.get("TELEGRAM_BOT_TOKEN", "your_token")

async def start(update: Update, context: ContextTypes.DEFAULT_TYPE):
    await update.message.reply_text(
        "🎾 歡迎使用網球場監控小助手！\n"
        "可用指令：\n"
        "/check 5/15-5/20 → 查詢時段\n"
        "/help → 查看說明"
    )

async def check_command(update: Update, context: ContextTypes.DEFAULT_TYPE):
    user_query = ' '.join(context.args)  # 如: "5/15-5/20"
    
    # 1. 解析日期範圍（簡易版，需實作 parse_date_range）
    # dates = parse_date_range(user_query) 
    dates = None # 測試用
    
    # 2. 单次抓取
    slots = scraper.fetch_venue_slots(target_dates=dates)
    
    # 3. AI 整理回覆
    reply = ai_helper.generate_telegram_reply(slots, user_query)
    
    # 4. 回傳（支援 Markdown）
    await update.message.reply_text(reply, parse_mode='Markdown')

if __name__ == "__main__":
    app = Application.builder().token(BOT_TOKEN).build()
    app.add_handler(CommandHandler("start", start))
    app.add_handler(CommandHandler("check", check_command))
    app.run_polling(allowed_updates=Update.ALL_TYPES)
```
